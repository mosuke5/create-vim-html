1 require 'logger'
 2 require 'docker'
 3 require 'optparse'
 4 
 5 args = {:expire_sec => '7200',
 6         :vimrc_path => nil,
 7         :docker_host => 'localhost',
 8         :docker_port => '4243',}
 9 
10 OptionParser.new do |opt|
11   opt.on('-s=SEC', '--expire-sec', 'default: 7200') { |v| args[:expire_sec] = v }
12   opt.on('-p=PATH', '--vimrc-path', 'もし指定しなければvimrcは消しません') { |v| args[:vimrc_path] = v }
13   opt.on('-H=HOST', '--docker-host', 'default: localhost') { |v| args[:docker_host] = v }
14   opt.on('-P=PORT', '--docker-port', 'default: 4243') { |v| args[:docker_port] = v }
15   opt.parse!(ARGV)
16 end
